Composition Sample

<div class="workspace">
  <div id="threeColumnHost"></div>
  <aside class="fixedColumn">Column 4</aside>
</div>

<style>
.workspace {
  display: grid;
  grid-template-columns: minmax(0, 1fr) 220px;
  gap: 10px;
  min-height: 340px;
}
</style>

const nestedHost = document.createElement("div");

createSplitter(threeColumnHost, {
  orientation: "horizontal",
  initialRatio: 0.28,
  minRatio: 0.18,
  maxRatio: 0.5,
  paneA: column1,
  paneB: nestedHost,
});

createSplitter(nestedHost, {
  orientation: "horizontal",
  initialRatio: 0.48,
  minRatio: 0.25,
  maxRatio: 0.75,
  paneA: column2,
  paneB: column3,
});

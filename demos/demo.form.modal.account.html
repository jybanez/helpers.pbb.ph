const accountModal = createAccountFormModal({
  fields: {
    name: "display_name",
    email: "contact_email",
  },
  extraRows: [
    [{ type: "text", content: "Keep contact details current for escalation routing." }],
    [{
      type: "input",
      input: "tel",
      name: "contact_number",
      label: "Contact Number",
    }],
  ],
  extraActionsPlacement: "start",
  extraActions: [{
    id: "change-password",
    label: "Change Password",
    onClick(values) {
      openChangePassword(values.contact_email);
      return false;
    },
  }],
});

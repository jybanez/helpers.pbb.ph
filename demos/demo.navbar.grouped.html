Grouped Navbar Demo

Open Operations to inspect the desktop grouped menu. Narrow the viewport to inspect the flattened mobile sections.

Library-Style App Chrome

createNavbar(host, {}, {
  brandText: "PBB Library",
  activeId: "collections",
  items: [
    { id: "home", label: "Home" },
    { id: "collections", label: "Collections" },
    { id: "my-library", label: "My Library" },
    {
      id: "operations",
      label: "Operations",
      menuGroups: operationGroups,
      menuOptions: { width: "viewport" },
    },
  ],
  onItemMenuSelect(item, menuItem, meta) {
    writeLog("navbar.grouped.select", {
      itemId: item.id,
      groupId: meta.group?.id,
      menuItemId: menuItem.id,
      source: meta.source,
    });
  },
});

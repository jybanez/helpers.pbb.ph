Audio Session Demo

Uses incident.media[] with recording_role=<role>-<call_id>-<timestamp>.

Sample Code

const session = createAudioCallSession(host, {
  caller: { name: "PBB Caller" },
  operator: { name: "Operator Reyes" },
  media: [{
    id: "caller-audio",
    type: "audio",
    processing: true,
    processingLabel: "Preparing caller audio...",
    peer_role: "caller",
    peer_label: "PBB Caller",
    created_at: "2026-04-29T00:00:00Z",
    metadata: { processing: true, track_kind: "audio" },
  }],
}, {
  baseUrl: "..",
});

session.update({
  media: [{
    id: "caller-audio",
    type: "audio",
    path: "samples/caller-final.wav",
    duration_seconds: 18,
    metadata: {
      recording_role: "caller-166-2026-04-29T00-00-00Z",
      track_kind: "audio",
    },
  }],
});

Usage

const strip = createHeartbeatStrip(host, {
  label: "Apas Source Hub",
  status: "online",
  ageSeconds: 42,
  history: hourlyBuckets,
  bucketCount: 48,
  rangeLabel: "last 48 hours",
});

strip.update({
  status: "stale",
  ageSeconds: 420,
  history: nextHourlyBuckets,
});

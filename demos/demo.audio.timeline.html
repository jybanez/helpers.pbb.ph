Audio Timeline Demo

Uses generic tracks[] and aligned segments[] rather than incident media rows.

Sample Code

const timeline = createAudioTimeline(host, {
  durationMs: 24000,
  tracks: [
    {
      id: "caller",
      label: "Caller",
      segments: [{ id: "caller-1", srcUrl: callerUrl, startOffsetMs: 0, durationMs: 12000 }],
    },
    {
      id: "operator",
      label: "Operator",
      segments: [{ id: "operator-1", srcUrl: operatorUrl, startOffsetMs: 1600, durationMs: 10000 }],
    },
    {
      id: "radio-net",
      label: "Radio Net",
      segments: [{ id: "radio-pending", processing: true, processingLabel: "Preparing radio channel..." }],
    },
  ],
}, {
  trackStyles: { "radio-net": "dots" },
});

timeline.update({
  durationMs: 24000,
  tracks: [{
    id: "radio-net",
    label: "Radio Net",
    segments: [{ id: "radio-pending", srcUrl: radioUrl, startOffsetMs: 0, durationMs: 24000 }],
  }],
});

Navbar Demo

Dedicated helper page for the shared top navigation bar. This page focuses on real item/action contracts instead of folding multiple navigation helpers into one manual. Narrow the viewport to inspect the viewport-aware hamburger behavior.

Preview

const navbar = createNavbar(host, {}, {
  brandText: "Hotline UI",
  brandSubtitle: "Build app-DWv_LvMM",
  brandMedia: iconShield,
  mobileCollapse: true,
  contentStart: "Operations",
  contentCenter: () => {
    const pill = document.createElement("span");
    pill.className = "demo-nav-pill";
    pill.innerHTML = 'Realtime Connected';
    return pill;
  },
  contentEnd: () => {
    const badge = document.createElement("span");
    badge.className = "demo-nav-badge";
    badge.textContent = "Region 7";
    return badge;
  },
  contentCenterMobile: { id: "status", label: "Realtime Connected", disabled: true },
  contentEndMobile: { id: "region", label: "Region 7", disabled: true },
  activeId: "dashboard",
  items: [
    { id: "dashboard", label: "Dashboard", icon: iconGrid },
    {
      id: "apps",
      label: "Apps",
      icon: iconAlert,
      menuItems: [
        { id: "app:relay", label: "Relay" },
        { id: "app:workspace", label: "Workspace" },
      ],
    },
    { id: "reports", label: "Reports", icon: iconBookmark },
  ],
  actions: [
    { id: "help", label: "Help", icon: iconHelp },
    {
      id: "profile",
      label: "Profile",
      icon: iconUser,
      menuItems: [
        { id: "manage", label: "Manage" },
        { id: "clear-cache", label: "Clear Cache" },
        { id: "logout", label: "Logout", danger: true },
      ],
    },
  ],
});

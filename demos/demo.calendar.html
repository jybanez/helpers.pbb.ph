Calendar Demo

Standalone month-grid primitive shared by date-related helpers.

Operational Month Selection

Use the primitive when a date surface should be visible inside a panel, drawer, dashboard, or custom app-owned workflow.

Single Date
Range Display 
createCalendar(document.getElementById("calendarSingle"), {
  value: "2026-07-24",
  viewDate: "2026-07-01",
  onSelect(date, state) {
    writeLog("calendar.single.select", { date: date.toISOString(), state });
  },
});

createCalendar(document.getElementById("calendarRange"), {
  mode: "range",
  viewDate: "2026-07-01",
  rangeStart: "2026-07-12",
  rangeEnd: "2026-07-18",
  disabledDates(date) {
    return date.getDay() === 0;
  },
  onSelect(date, state) {
    writeLog("calendar.range.select", { date: date.toISOString(), state });
  },
});

Community Recognition Badges

Sample Code

const badge = document.createElement("span");
badge.className = "ui-badge is-success";
badge.textContent = "Verified";

const icon = createIcon("community.badge", {
  size: 14,
  title: "Badge",
  decorative: false,
});
badge.prepend(icon);

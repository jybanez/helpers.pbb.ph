Activity Chart Demo

Reusable project/day activity chart for shared coordination logs.

Matrix

import { uiLoader } from "../js/ui/ui.loader.js";

await uiLoader.loadMany(["ui.activity.chart"]);
const createActivityChart = await uiLoader.get("ui.activity.chart");

const chart = createActivityChart(document.getElementById("activityHost"), records, {
  projects,
  dates,
  mode: "matrix",
  metric: "total",
  showHeader: false,
  projectOrder: "input",
  onSelect({ project, date, summary }) {
    filterTimelineByProjectDay(project, date, summary);
  },
  onHover({ project, date, summary }) {
    previewActivity(project, date, summary);
  },
});

chart.setMetric("direct");
chart.setMode("sparklines");

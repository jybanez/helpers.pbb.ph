Usage

const timer = createElapsedTime(host, {
  startTime: incident.reported_at,
  ariaLabel: "Incident active duration",
  thresholds: [
    { atMs: 30 * 60 * 1000, variant: "warn" },
    { atMs: 2 * 60 * 60 * 1000, variant: "danger" },
  ],
});

Usage

const objectLayer = createGameObjectLayer();
const card = createFlipCard({
  id: "card-1",
  x: 80,
  y: 80,
  frontLabel: "A",
  onSelect(card) {
    revealAndCheckMatch(card);
  },
  renderFront(ctx, card) {
    drawCardArt(ctx, card);
  },
});

objectLayer.add(card);
createPointerInputRouter(canvas, objectLayer);

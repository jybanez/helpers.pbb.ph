Operational metric cards for active incidents, SITREPs, and support dashboards.

Usage

const stats = createStatCards(host, [
  {
    id: "risk",
    label: "People at Risk",
    value: 61,
    unit: "people",
    note: "Across 5 source hubs",
    icon: "population.people-at-risk",
    tone: "warning",
    trend: { direction: "up", label: "+12" },
  },
], {
  selectable: true,
  onSelect(item) {
    openMetric(item.id);
  },
});

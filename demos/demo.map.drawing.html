Usage

const maplibregl = await ensureMapLibre(mapConfig);
const map = new maplibregl.Map({
  container: mapNode,
  style,
  center: [123.8854, 10.3157],
  zoom: 13,
});

const drawing = createMapDrawingTools(container, {
  modes: ["select", "point", "line", "polygon"],
  activeMode: "select",
  adapter: createMapLibreDrawingAdapter(map),
  onCreate(feature) {
    saveFeature(feature);
  },
});

Dedicated helper page for the shared password input used by login, re-auth, and any other narrow password-entry flow.

Standalone Password Field

const password = createPasswordField(host, {
  id: "operator-password",
  name: "operator_password",
  placeholder: "Enter password",
  autocomplete: "current-password",
  onChange(value) {
    writeLog("password.change", { length: value.length });
  },
});
